/**
 * @module root
 * Início da API, conexão com porta e requerimento de pacotes
 */
const express = require('express');
const app = express();
const bodyParser = require('body-parser');
const config = require('./config.json');
const port = config.porta;
const got = require("got");

app.use(bodyParser.urlencoded({
    extended: true
}));

app.use(bodyParser.json());

const router = express.Router();
router.get('/', (req, res) => res.json({
    message: 'Funcionando'
    
}));

/**
 * Rota para receber artigo da NASA
 */
router.get('/nasa', (req, res) => {
    got(config.urlpesquisaNasa, {
            json: true
        })
        .then(response => {
            console.log(response.body.url);
            console.log(response.body.explanation);
            console.log({
                "response": response.body
            });
            res.send(response.body);
        })
        .catch(error => {
            console.log(error.response.body);
        });
});

/**
 * Rota para receber notícias do wordpress da Febrasgo
 */
router.get('/noticias',(req, res) => {
    let urlChamada = config.urlWordPressLocal + config.expressao
    got( urlChamada  , {
        json: true
    })
    .then(response => {        
        console.log({
            "response": response.body
        });
        res.send(response.body);
    })
    .catch(error => {
        console.log(error.response);
    });
});


app.use('/', router);
app.listen(port);
